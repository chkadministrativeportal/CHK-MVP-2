var isOpen = false;
var nomineeIdx = null;
var COLORS = ['#F59E0B','#EF4444','#3B82F6','#10B981','#8B5CF6','#F97316','#fff'];

function checkPeriod() {
var now = new Date();
isOpen = now >= CONFIG.nominationStart && now <= CONFIG.nominationEnd;
var dot = document.getElementById('statusDot');
var txt = document.getElementById('statusText');
var banner = document.getElementById('closedBanner');
var m = now.getMonth() + 1;
if (m >= 1 && m <= 4) document.getElementById('term1').classList.add('active');
else if (m >= 5 && m <= 8) document.getElementById('term2').classList.add('active');
else document.getElementById('term3').classList.add('active');
if (isOpen) {
dot.classList.remove('off');
var end = CONFIG.nominationEnd.toLocaleDateString('en-PH', {month:'long', day:'numeric', year:'numeric'});
txt.textContent = 'NOMINATIONS OPEN - Closes ' + end;
banner.style.display = 'none';
} else {
dot.classList.add('off');
txt.textContent = CONFIG.termLabel + ' - Nominations Closed';
banner.style.display = 'block';
}
}

function buildTabs() {
var wrap = document.getElementById('tabsContainer');
wrap.innerHTML = '';
for (var i = 0; i < TABS.length; i++) {
(function(t, idx) {
var btn = document.createElement('button');
btn.className = 'tab' + (idx === 0 ? ' active' : '');
btn.textContent = t.label;
btn.onclick = function() { switchTab(t.key, btn); };
wrap.appendChild(btn);
})(TABS[i], i);
}
}

function buildGrid(filter) {
var grid = document.getElementById('staffGrid');
grid.innerHTML = '';
var list = [];
for (var i = 0; i < STAFF.length; i++) {
if (filter === 'all' || STAFF[i].category === filter) list.push(STAFF[i]);
}
for (var j = 0; j < list.length; j++) {

function switchTab(cat, btn) {
var tabs = document.querySelectorAll('.tab');
for (var i = 0; i < tabs.length; i++) tabs[i].classList.remove('active');
btn.classList.add('active');
buildGrid(cat);
}

function openModal(idx, e) {
if (e) e.stopPropagation();
if (!isOpen) return;
var s = STAFF[idx];
if (!s) return;
if (s.ineligible) {
var cards = document.querySelectorAll('.card');
for (var i = 0; i < cards.length; i++) {
if (parseInt(cards[i].getAttribute('data-idx')) === idx) {
cards[i].classList.remove('shaking');
void cards[i].offsetWidth;
cards[i].classList.add('shaking', 'benched');
setTimeout(function(c) { return function() { c.classList.remove('shaking'); }; }(cards[i]), 600);
}
}
document.getElementById('benchName').textContent = s.name;
document.getElementById('benchModal').classList.add('open');
return;
}
nomineeIdx = idx;
document.getElementById('modalAvatar').src = s.img;
document.getElementById('modalName').textContent = s.name;
document.getElementById('modalRole').textContent = s.role + ' - ' + s.section;
document.getElementById('nominatorName').value = '';
document.getElementById('nominatorClass').value = '';
document.getElementById('nominationReason').value = '';
document.getElementById('modalOverlay').classList.add('open');
}

function closeModal() {
document.getElementById('modalOverlay').classList.remove('open');
nomineeIdx = null;
}

document.getElementById('benchModal').classList.remove('open');
}

function submitNomination() {
var nominatorName = document.getElementById('nominatorName').value.trim();
var nominatorClass = document.getElementById('nominatorClass').value;
var reason = document.getElementById('nominationReason').value.trim();

// Validate all fields
var valid = true;
if (!nominatorName) {
document.getElementById('nominatorName').style.borderColor = '#EF4444';
setTimeout(function() { document.getElementById('nominatorName').style.borderColor = 'rgba(255,255,255,.1)'; }, 1500);
valid = false;
}
if (!nominatorClass) {
document.getElementById('nominatorClass').style.borderColor = '#EF4444';
setTimeout(function() { document.getElementById('nominatorClass').style.borderColor = 'rgba(255,255,255,.1)'; }, 1500);
valid = false;
}
if (!reason) {
document.getElementById('nominationReason').style.borderColor = '#EF4444';
setTimeout(function() { document.getElementById('nominationReason').style.borderColor = 'rgba(255,255,255,.1)'; }, 1500);
valid = false;
}
if (!valid) return;

var s = STAFF[nomineeIdx];
var submitBtn = document.getElementById('submitBtn');
submitBtn.textContent = 'SUBMITTING...';
submitBtn.disabled = true;

var payload = {
nominatorName: nominatorName,
classification: nominatorClass,
nomineeName: s.name,
nomineeSection: s.section,
reason: reason,
term: CONFIG.termLabel
};

fetch(CONFIG.appsScriptUrl, {
method: 'POST',
mode: 'no-cors',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify(payload)
})
.then(function() {
// Mark card as nominated
var cards = document.querySelectorAll('.card');
for (var i = 0; i < cards.length; i++) {
if (parseInt(cards[i].getAttribute('data-idx')) === nomineeIdx) cards[i].classList.add('nominated');
}
closeModal();
launchConfetti();
showStamp();
showToast('Boto para kay ' + s.name + ' - Salamat, ' + nominatorName + '!');
})
.catch(function() {
closeModal();
launchConfetti();
showStamp();
showToast('Boto para kay ' + s.name + ' - Salamat!');
})
.finally(function() {
submitBtn.textContent = 'SUBMIT VOTE';
submitBtn.disabled = false;
});
}

function launchConfetti() {
for (var i = 0; i < 70; i++) {
(function(i) {
setTimeout(function() {
var p = document.createElement('div');
p.className = 'confetti-piece';
p.style.left = (Math.random() * 100) + 'vw';
p.style.top = '-10px';
p.style.background = COLORS[Math.floor(Math.random() * COLORS.length)];
p.style.width = (6 + Math.random() * 8) + 'px';
p.style.height = (6 + Math.random() * 8) + 'px';
p.style.borderRadius = Math.random() > 0.5 ? '50%' : '2px';
p.style.animationDuration = (1.2 + Math.random() * 1.2) + 's';
p.style.animationDelay = (Math.random() * 0.4) + 's';
document.body.appendChild(p);
setTimeout(function() { if (p.parentNode) p.parentNode.removeChild(p); }, 2600);
}, i * 18);
})(i);
}
}
